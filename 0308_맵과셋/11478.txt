#include <iostream>
#include <string>
#include <set>

using namespace std;

int main()
{
    //입출력 속도 향상
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    string input,substr;  //입력되는 문자열, 부분 문자열
    set<string> s; //서로 다른 문자열 저장할 set
    cin>>input;
    
    int length = input.length(); //입력된 문자열의 길이
    
    /*
    (i,j)          
    00
    10 11
    20 21 22
    30 31 32 33
    40 41 42 43 44
    
    -> (j,i)
    00 01 02 03 04
    11 12 13 14
    22 23 24
    33 34
    44 
    */
    for (int i=0; i<length; i++){ 
        for(int j=0; j<i+1; j++){
        substr = input.substr(j,i); //string 함수로 부분 문자열을 구하기
        s.insert(substr); // set에 삽입(서로다른 문자열만 저장됨)
        }
    }
    
    cout<<s.size()<<'\n'; //set에 저장되어 있는 문자열 개수
    
    
    return 0;
}
